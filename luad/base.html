enum LuaType ;
Enumerates all Lua types.

Nil nil ;
Special value representing the Lua type and value ' nil '.

Examples:
Useful for clearing keys in a table: 
	lua["n"] = 1.23;
	assert(lua.get!double("n") == 1.23);

	lua["n"] = 1.23;
	assert(lua["n"].type == LuaType.Nil);
